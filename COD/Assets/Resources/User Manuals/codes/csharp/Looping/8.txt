using System;

public class Program {
    public static void Main(string[] args) {
        for (int num = 1; num <= 10; num++) {
            if (num == 5 || num == 6) {
                /* if this statement is executed it will ignore the print 
                   statement below and the execution jumps back to the loop */
                continue;
            }

            Console.Write(num + " ");
        }
    }
}
