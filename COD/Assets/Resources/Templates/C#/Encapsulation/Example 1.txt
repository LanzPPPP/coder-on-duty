using System;

abstract class Animal
{
    public abstract string Name
    {
        get;
    }

    public void DisplayName()
    {
        Console.WriteLine($"Animals name is {Name}.");
    }
}

class Dog : Animal
{
    public override string Name
    {
        get
        {
            return "Dog";
        }
    }
}

class Cat : Animal
{
    public override string Name
    {
        get
        {
            return "Cat";
        }
    }
}

public class Program
{
    public static void Main(string[] args)
    {
        Animal dog = new Dog();
        Animal cat = new Cat();

        dog.DisplayName();
        cat.DisplayName();
    }
}
